##SQLDeveloper(Tables)
1. TABLE COMMENT
CREATE TABLE COMMENT (
  ID NUMBER,
  CONTENT NVARCHAR2(2000),
  REGDATE TIMESTAMP(6),
  WRITER_ID NVARCHAR(50),
  NOTICE_ID NUMBER
)

2. TABLE MEMBER
CREATE TABLE MEMBER (
  ID NVARCHAR(50),
  PWD VARCHAR2(50),
  NAME VAWRCHAR2(50),
  GENDER NCHAR(2),
  BIRTHDAY CHAR(10),
  PHONE CHAR(13),
  REGDATE DATE,
  EMAIL VARCHAR2(200)
)

3. TABLE MEMBER_ROLE (
  MEMBER_ID NVARCHAR2(50),
  ROLE_ID VARCHAR2(50)
)

4. TABLE NOTICE (
  ID NUMBER,
  TITLE NVARCHAR2(100),
  WRITER_ID NVARCHAR2(50),
  CONTENT CLOB,
  REGDATE TIMESTAMP(6),
  HIT NUMBER,
  FILES NVARCHAR2(1000),
  PUB NUMBER(1,0)
)

5. TABLE ROLE(
  ID VARCHAR2(50),
  DESCRIPTION NVARCHAR2(500)
)

##VIEW
CREATE VIEW NOTICE_VIEW
AS
SELECT * FROM (
  SELECT ROWNUM NUM, N.* FROM (
    SELECT * FROM NOTICE ORDER BY REGDATE DESC
  )
);

SELECT * FROM NOTICE_VIEW
WHERE NUM BETWEEN ? AND ?;
